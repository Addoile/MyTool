VSHVAR VSAPI vsHvarNextEl(VSHVAR hvarArrayOrHashTab, HVAR hvarstart)

Parameters:

hvar - Handle to hash table or array variable.

hvarIndex - Handle to index variable.

Returns: VSHVAR

Returns reference to next element in array or hash table. If index is empty (see vsHvarMakeEmpty()), the first element is returned. If there are no more elements, index is set to empty and 0 is returned.

Example:

VSHVAR  hvar;
VSHVAR hvarel;

// Create hash table with ["a"]=1, ["b"]=2, and ["c"]=3
hvar=vsHvarAlloc();
hvarel=vsHvarHashTabEl(hvar,"a");vsHvarSetI(hvarel,1);
hvarel=vsHvarHashTabEl(hvar,"b");vsHvarSetI(hvarel,2);
hvarel=vsHvarHashTabEl(hvar,"c");vsHvarSetI(hvarel,3);
// Traverse the elements in hash table
VSHVAR hvarIndex;
for (vsHvarMakeEmpty(hvarIndex);;) {
   VSHVAR    hvarel;
    hvarel=vsHvarNextEl(hvar,hvarIndex);
    if (!hvarel) break;
    vsMessageBox ((char *)vsHvarGetLstr (hvarel)->str);
}
