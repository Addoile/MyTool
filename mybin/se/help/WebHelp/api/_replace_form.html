typeless _mfmore.on_create()

Displays the Replace dialog box.

Returns: typeless

Returns '' if dialog box is cancelled. Otherwise, 1 is returned and the global variables _param1-_param5 are set as follows:

_param1
Search string.
_param2
Replace string.
_param3
Search options.
_param4
Files to search through.
_param5
Buffers to search through.

Example:

_str show('-modal _replace_form') 

_command gui_replace()
        name_info(','VSARG2_ EDITORCTL)
{
   _macro_delete_line()
   result=show('-modal _replace_form',arg(1))
   if (result=='') {
      return(COMMAND_CANCELLED_RC);
   }
   _macro('m',_macro('s'))
   if (_param4=='' && _param5=='') {
      _macro_call('replace2', _param1, _param2, _param3)
      return(gui_replace2(_param1, _param2, _param3))
   } else {
      if (substr(_param5,1,1)=='@' && _macro()) {
         _message_box(nls("Sorry, can't generate macro code for this
operation."));
      } else {
         _macro_call('_mfreplace', _param1, _param2, _param3,
_param4, _param5)
      }
      return(_mfreplace(_param1, _param2, _param3, _param4,
_param5));
   }
}
