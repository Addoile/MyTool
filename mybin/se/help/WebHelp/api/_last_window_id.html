int _last_window_id()

Returns: int

Returns the highest possible window id allocated. This function is used in conjunction with the _iswindow_valid function to loop through all windows.

Example:

_find_formobj(form_name /*,'E' - Find edited forms only | 'N' - Find non-
edited forms only */)
{
   last=_last_window_id()
   option=upcase(arg(2))
   for (i=1;i<=last;++i) {
      if (_iswindow_valid(i) && i.p_object==OI_FORM &&
           (option=='' || (option=='E' && i.p_edit) ||
                   (option=='N' && !i.p_edit)
           )
         ) {
         if (name_eq(i.p_name,form_name)) {
            return(i)
         }
      }
   }
   return(0)
}
